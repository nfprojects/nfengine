# @brief  CMake for TestCalcLib

MESSAGE(STATUS "Generating build files for testCalcLib")

SET(TESTCALCLIB_SOURCES
    TestCalcLib.cpp
)

SET(TESTCALCLIB_HEADERS
    TestCalcLib.hpp
)

# build shared lib
ADD_LIBRARY(testCalcLib SHARED ${TESTCALCLIB_SOURCES} ${TESTCALCLIB_HEADERS})

TARGET_LINK_DIRECTORIES(testCalcLib
    PRIVATE ${NFCOMMONTEST_DIRECTORY}
    PRIVATE ${NFE_OUTPUT_DIRECTORY}
)

TARGET_COMPILE_DEFINITIONS(testCalcLib
    PRIVATE TESTCALCDLL_EXPORTS
)

SET_PROPERTY(TARGET testCalcLib PROPERTY FOLDER Src/Tests/TestResources)
NFE_SOURCE_GROUP_BY_DIR(TESTCALCLIB_SOURCES)
NFE_SOURCE_GROUP_BY_DIR(TESTCALCLIB_HEADERS)
